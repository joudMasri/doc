import { Injectable } from '@angular/core';
import { Day, CalendarEvent } from '../models/events';

@Injectable({
  providedIn: 'root'
})
export class DateService {

  constructor() { }

  getMonthDays(
    month: number,
    year: number,
    minDate: {
      day: number,
      month: number,
      year: number
    },
    maxDate: {
      day: number,
      month: number,
      year: number
    },
    events: CalendarEvent[]
  ): Day[] {
    let monthDays: Day[] = [];
    monthDays = [...this.fillStartGap(month, year)];

    for (let i = 1; i <= this.getDaysCount(month, year); i++) {
      let day: Day = {
        number: i,
        enabled: this.isEnabled(i, month, year, minDate, maxDate),
        month: month,
        year: year,
        events: this.getDayEvents(i, month, year, events)
      };
      monthDays.push(day);
    }

    monthDays =[ ...monthDays, ...this.fillEndGap(monthDays.length, month, year)];

    return monthDays;
  }

  fromObjectToDateString(object: {day?: number, month?: number, year?: number}) {
    return new Date(Date.UTC(object?.year || 0, object?.month || 0, object?.day || 0, 0, 0, 0, 0))
  }

  private isEnabled(day: number, month: number, year: number,
    minDate: {day: number, month: number, year: number},
    maxDate: {day: number, month: number, year: number}): boolean {
    const currentDate = new Date(year, month, day);
    const min = this.fromObjectToDateString(minDate);
    const max = this.fromObjectToDateString(maxDate);
    return currentDate <= max && currentDate >= min
  }

  private getPreviousMonthDays(requiredDaysCount: number, prevMonth: number, year: number) {
    let array: Day[] = [];

    let i = requiredDaysCount;
    let daysCount = this.getDaysCount(prevMonth, year);

    while (array.length < i) {
      array.push({ number: daysCount, enabled: false, month: prevMonth, year: year, events: [] });
      daysCount--;
    }

    return array.reverse();
  }

  getDaysCount(month: number, year: number) {
    return new Date(year, month + 1, 0).getDate();
  }

  private getFirstDay(month: number, year: number) {
    return new Date(year, month, 1).getDay();
  }

  getDayEvents(day: number, month: number, year: number, calendarEvents: CalendarEvent[]): CalendarEvent[] {
    let events: CalendarEvent[] = [];
    calendarEvents.forEach(event => {
      if (event.startDate?.getDate() === day && event.startDate.getMonth() === month && event.startDate.getFullYear() === year) {
        events.push(event);
      }
    });
    return events;
  }

  private fillStartGap(month: number, year: number) : Day[] {
    if (this.getFirstDay(month, year) !== 0) {
      return this.getPreviousMonthDays(this.getFirstDay(month, year), month - 1, year);
    } else return [];
  }

  private fillEndGap(length: number, month: number, year: number) : Day[] {
    let days: Day[] = [];

    if (length % 7 !== 0) {
      let requiredDays = 7 - length % 7;
      for (let i = 1; i <= requiredDays; i++) {
        let events: CalendarEvent[] = [];
        let day = { number: i, enabled: false, month: month + 1, year: year, events: events };
        days.push(day);
      }
    }

    return days;
  }

  getRangeEnd(day: number, month: number, year: number,
    startDate: Date,
    endDate: Date
  ) {
    const startDateObject = this.fromDateStringToObject(startDate);
    const endDateObject = this.fromDateStringToObject(endDate)
    return (day === startDateObject.day && month === startDateObject.month && year === startDateObject.year)
          || (day === endDateObject.day && month === endDateObject.month && year === endDateObject.year)
  }

  fromDateStringToObject(date: Date) {
    return {day: date.getDate(), month: date.getMonth(), year: date.getFullYear()}
  }
  getInRange(day: number, month: number, year: number,
    start: Date,
    end: Date
  ) : boolean {
    const currentDate = new Date(year, month, day, 0, 0);
    return currentDate <= end && currentDate >= start;
  }

  weekDays = [
    'Sun',
    'Mon',
    'Tue',
    'Wed',
    'Thu',
    'Fri',
    'Sat',
  ]

  months = [
    "Jan",
    "Feb",
    "Mar",
    "Apr",
    "May",
    "Jun",
    "Jul",
    "Aug",
    "Sep",
    "Oct",
    "Nov",
    "Dec"
  ]

  longMonths = [
    "January",
    "February",
    "March",
    "April",
    "May",
    "June",
    "July",
    "August",
    "September",
    "October",
    "November",
    "December"
  ]
}
