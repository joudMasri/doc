import { Component, EventEmitter, HostListener, Input, OnInit, Optional, Output, Self, ViewChild, AfterContentInit } from '@angular/core';
import { faChevronDown, faCheck } from '@fortawesome/free-solid-svg-icons';
import { ControlValueAccessor, NgControl, Validators } from '@angular/forms';
import { toggle } from '../../../animations/toggle.animation';
@Component({
  selector: 'lib-checkbox',
  templateUrl: './checkbox.component.html',
  styleUrls: ['./checkbox.component.scss'],
  animations: [toggle]
})
export class CheckboxComponent implements OnInit, AfterContentInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }
  ngAfterContentInit(): void {
    if (this.selectedOptions.length) {
      let selections = this.selectedOptions;
      this.selectedOptions = [];
      selections.forEach(element => {
        this.toggleValue(element)
      });
    } else {
      this.initialSelections.forEach(element => {
        this.toggleValue(element)
      });
    }
  }

  ngAfterViewInit(): void {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      this.control.control?.addValidators([Validators.minLength(this.min)])
    }
  }

  ngOnInit(): void {}

  expanded = false;
  touched = false;

  faChevronDown = faChevronDown;
  faCheck = faCheck;

  @ViewChild('dropdownRef') dropdownRef : any;
  @ViewChild('listRef') listRef: any;

  @Output('onValueChange') valueEmitter = new EventEmitter<any>();
  @Input('dropdown') dropdown = true;
  @Input('name') name: string | null = null;
  @Input('class') class: string | null = null;
  @Input('placeholder') placeholder = 'Select an option';
  @Input('label') label: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('required') required = true;
  @Input('min') min = 1;
  @Input('absolute') absolute = false;
  @Input('disabled') disabled = false;
  @Input('sm') small = false;
  @Input('autofocus') autofocus = false;
  @Input('validationMessage') validationMessage = true;
  @Input('disableAll') disableAll = false;
  @Input('isReactive') isReactive = true;
  @Input('options') options: any = [];

  @Input('initialSelections') initialSelections: any[] = [];

  selectedOptions: any[] = [];

  get invalid() {
    return this.isReactive ? this.control.touched && this.control.invalid && !this.disabled : 
    this.touched && this.required && this.selectedOptions.length < this.min && !this.disabled;
  }

  get isValid() {
    return this.required && this.selectedOptions.length < this.min;
  }

  selectAll = () => {
    this.touched = true;
    if (this.selectedOptions.length === this.options.length) {
      this.selectedOptions = []
    } else {
      this.selectedOptions = [...this.options]
    }

    if (this.control) {
      this.onTouch();
      this.onModelChange(this.selectedOptions);
    } else {
      this.valueEmitter.emit(this.selectedOptions);
    }
  };

  isSelected(option: any) {
    return this.selectedOptions.findIndex(o => o.value.toString() === option.value.toString()) >= 0;
  }
  toggleValue = (option: {label: string, value: any}) => {
    this.touched = true;
    if (this.isSelected(option)) {
      this.selectedOptions =
        this.selectedOptions.filter(o => o.value !== option.value);
    } else {
      this.selectedOptions = [...this.selectedOptions, option];
    }

    if (this.control) {
      this.onTouch();
      this.onModelChange(this.selectedOptions);
    } else this.valueEmitter.emit(this.selectedOptions);
  }


  @HostListener('document:click', ['$event'])
  documentClick(event: any) {
    if (!this.dropdownRef?.nativeElement?.contains(event.target)
    && !this.listRef?.nativeElement?.contains(event.target)
    && !event.target.classList.contains('fa-check'))
      this.expanded = false;
  }
  
  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler() {
    this.expanded = false;
  }

  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {
    this.selectedOptions = obj || [];
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }
}

    [@.disabled]="!dropdown"
    [class.mt-2]="!dropdown"

    <div class="checklist">
      <div *ngIf="!disableAll" class="option all">
        <div
          class="box"
          tabindex="0"
          [class.checked]="selectedOptions.length === options.length"
          (click)="selectAll()"
          (keydown.enter)="selectAll()"

          </fa-icon>
        </div>
        <span>All</span>

          (click)="toggleValue(option)"

      * Select at least {{ min }} <span [innerText]="min === 1? 'option' : 'options'"></span>.
    </div>
    <ng-content></ng-content>
  </small>
