import { Component, Input, OnInit, Optional, Output, Self, EventEmitter, ViewChild, ElementRef } from '@angular/core';
import { faClock } from '@fortawesome/free-regular-svg-icons';
import { faAngleDown, faCheck } from '@fortawesome/free-solid-svg-icons';
import { ControlValueAccessor, NgControl } from '@angular/forms';

@Component({
  selector: 'lib-time-picker',
  templateUrl: './time-picker.component.html',
  styleUrls: ['./time-picker.component.scss'],
})
export class TimePickerComponent implements OnInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }

  faClock = faClock;
  faAngleDown = faAngleDown;
  faCheck = faCheck;

  ngOnInit(): void {}

  not = true;
  allDay = true;
  focused = false;

  @ViewChild('inputRef') inputRef!: ElementRef;
  @ViewChild('componentRef') componentRef!: ElementRef;
  @ViewChild('boxRef') boxRef!: ElementRef;

  @Input('label') label: string | null = null;
  @Input('required') required = true;
  @Input('disabled') disabled = false;
  @Input('autofocus') autofocus = false;
  @Input('autocomplete') autocomplete = false;
  @Input('name') name: string | null = null;
  @Input('max') max = '16:00';
  @Input('min') min = '09:00';
  @Input('class') class: string | null = null;
  @Input('value') value: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('sm') small = false;

  @Output('onValueChange') onValueChange = new EventEmitter();

  switch() {
    this.allDay = !this.allDay;
    this.inputRef.nativeElement.value = null;
  }
  emitDate() {
    let value: string;
    if (this.allDay) {
      value = 'All Day';
    } else {
      value = this.inputRef.nativeElement.value;
    }

    if (this.control) {
      this.onTouch();
      this.onModelChange(value)
    } else this.onValueChange.emit(value)
  }


  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {
    this.value = obj;
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

}
