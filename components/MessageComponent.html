import { Component, Input, OnInit, Output, EventEmitter } from '@angular/core';
import { faTimes } from '@fortawesome/free-solid-svg-icons';
import { fade } from '../../../animations/shared.animation';

@Component({
  selector: 'lib-message',
  templateUrl: './message.component.html',
  styleUrls: ['./message.component.scss'],
  animations: [fade]
})
export class MessageComponent implements OnInit {

  constructor() { }

  ngOnInit(): void {
  }

  faTimes = faTimes;

  visible = false;

  @Input('title') title = 'Title';
  @Input('duration') duration = 5000;
  @Input('sticky') sticky = false;
  @Input('context') context = 'primary';
  @Input('clickable') clickable = false;
  @Output('onClose') onCloseEmitter = new EventEmitter();
  @Output('onClick') onClickEmitter = new EventEmitter();

  open() {
    this.visible = true;
    if (!this.sticky) {
      setTimeout(() => this.visible = false, this.duration)
    }
  }

  close() {
    this.visible = false;
    this.onCloseEmitter.emit();
  }

  click() {
    if (this.clickable) {
      this.onClickEmitter.emit();
    }
  }

}
