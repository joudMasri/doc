import { TitleCasePipe } from '@angular/common';
import { Component, ElementRef, EventEmitter, HostListener, Input, OnInit, Optional, Output, Self, ViewChild, AfterViewInit, ChangeDetectorRef } from '@angular/core';
import { PhonePipe } from '../../../helpers/pipes/phone.pipe';
import { StringPipe } from '../../../helpers/pipes/string.pipe';
import { Country } from '../../../models/country';
import { ControlValueAccessor, NgControl, Validators } from '@angular/forms';
import { toggle } from '../../../../lib/animations/toggle.animation';
import { countries } from '../../../../lib/data/countries';
import { phones } from '../../../../lib/data/codes';
import { isNumberLengthValid } from '../../../helpers/validators/phone-validators';

@Component({
  selector: 'lib-phone',
  templateUrl: './phone.component.html',
  styleUrls: ['./phone.component.scss'],
  providers: [PhonePipe, TitleCasePipe, StringPipe],
  animations: [toggle]
})
export class PhoneComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(
    private titleCasePipe: TitleCasePipe,
    private stringPipe: StringPipe,
    private phonePipe: PhonePipe,
    private cdRef: ChangeDetectorRef,
    @Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }

  ngOnInit(): void {}
  expanded = false;
  touched = false;
  focus = false;
  countries: Country[] = [...countries];
  phones: any = phones;

  @ViewChild('dropdownRef') dropdownRef!: ElementRef;
  @ViewChild('searchInputRef') searchInputRef!: ElementRef;
  @ViewChild('phoneNumberRef') phoneNumberRef!: ElementRef;
  @Input() value: {country: Country, number: any} = {
    country: { code: "LB", name: "Lebanon" },
    number: { prefix: 961, body: '' }
  };

  toggle() {
    this.expanded = !this.expanded;
    this.countries = countries;
  }

  @Output('onValueChange') valueEmitter = new EventEmitter<any>();
  requiredLength = 8;
  @Input('sm') small = false;
  @Input('label') label: string | null = 'test';
  @Input('labelTop') labelTop = false;
  @Input('class') class: string | null = null;
  @Input('required') required = true;
  @Input('absolute') absolute = false;
  @Input('border') border = true;
  @Input('disabled') disabled = false;
  @Input('bold') bold = false;
  @Input('virtualScroll') virtualScroll: boolean | null = true;
  @Input('animation') animation = true;
  @Input('autocomplete') autocomplete = false;
  @Input('name') name: string | null = null;
  @Input('isReactive') isReactive = true;

  get invalid() {
    return this.isReactive ? this.control.touched && this.control.invalid && !this.disabled :
    this.touched && this.phoneNumberRef.nativeElement.value >= 8 && !this.disabled;
  }

  ngAfterViewInit(): void {
    this.phoneNumberRef.nativeElement.value = this.value.number.body;
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      this.control.control?.addValidators(Validators.minLength(this.requiredLength));
    }
    this.cdRef.detectChanges();
  }


  emitCodeValue(country: Country) {
    this.touched = true;
    this.value.country = country;
    const code = this.phones[country.code];

    if (+code === 961) {
      this.phoneNumberRef.nativeElement.value =
      this.phoneNumberRef.nativeElement.value.substring(0, 10);
    }

    this.value.country = country;
    this.value.number.prefix = code;

    if (this.control) {
      this.onTouch();
      this.onModelChange(this.phoneNumberRef.nativeElement.value);
    }
    else this.valueEmitter.emit(this.phoneNumberRef.nativeElement.value);

    this.expanded = false;
  }

  emitPhoneNumber() {
    this.touched = true;
    this.value.number.body =
    this.phonePipe.transform(this.phoneNumberRef.nativeElement.value,
      this.pipeArgs);
    this.phoneNumberRef.nativeElement.value = this.value.number.body;
    if (this.control) {
      this.onTouch();
      this.onModelChange(this.formatPhoneNumber(this.value));
    }
    else this.valueEmitter.emit(this.formatPhoneNumber(this.value));
  }

  get pipeArgs() {
    let maximum = 15 - this.value.country.code.replace(/\s/g, "").length;
    return this.value.country.code === 'LB' ? 8 : maximum
  }

  refill() {
    this.expanded = false;
    this.countries = countries;
  }
  filter($event: any) {
    this.expanded = true;
    const transformation = this.titleCasePipe.transform(this.stringPipe.transform($event.target.value));
    this.countries = this.countries.filter(country => country.name.includes(transformation));
  }

  pickFirst() {
    this.emitCodeValue(this.countries[0]);
  }

  formatPhoneNumber(value: {country: Country, number: any}) : string {
    let fOutput = `${value.number.prefix}${value.number.body}`
    let output = fOutput.replace(/\s/g, "");
    return `+${output}`
  }

  @HostListener('document:click', ['$event'])
  documentClick(event: any) {
    if (!this.dropdownRef?.nativeElement?.contains(event.target) &&
    !this.searchInputRef?.nativeElement.contains(event.target)) {
      this.expanded = false;
    }
  }

  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler() {
    this.expanded = false;
  }

  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {

      this.value = {
        country: { code: "LB", name: "Lebanon" },
        number: { prefix: 961, body: obj }
      };

  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }
}
