import { TitleCasePipe } from '@angular/common';
import { Component, EventEmitter, HostListener, Input, OnInit, Optional, Output, Self, ViewChild, ElementRef, AfterViewInit } from '@angular/core';
import { NgControl, ControlValueAccessor, Validators } from '@angular/forms';
import { Country } from '../../../models/country';
import { toggle } from '../../../animations/toggle.animation';
import { StringPipe } from '../../../helpers/pipes/string.pipe';
import { countries } from '../../../../lib/data/countries';

@Component({
  selector: 'lib-countries',
  templateUrl: './countries.component.html',
  styleUrls: ['./countries.component.scss'],
  providers: [TitleCasePipe, StringPipe],
  animations: [toggle]
})
export class CountriesComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(
    private titleCasePipe: TitleCasePipe,
    private stringPipe: StringPipe,
    @Self() @Optional() public control: NgControl
  ) {
    if (this.control) this.control.valueAccessor = this;
  }

  ngAfterViewInit(): void {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
    }
  }

  ngOnInit(): void {
    this.selectValue(this.selectedOption);
  }

  expanded = false;
  touched = false;
  focus = false;
  countries: Country[] = [...countries];

  @ViewChild('dropdownRef') dropdownRef!: ElementRef;
  @ViewChild('inputRef') inputRef!: ElementRef;

  @Input('selectedOption') selectedOption = { code: "LB", name: "Lebanon" };
  @Output('onValueChange') valueEmitter = new EventEmitter<any>();
  @Input('label') label: string | null = null;
  @Input('disabled') disabled = false;
  @Input('labelTop') labelTop = false;
  @Input('class') class: string | null = null;
  @Input('sm') small = false;
  @Input('required') required = true;
  @Input('name') name: string | null = null;
  @Input('absolute') absolute = false;
  @Input('border') border = true;
  @Input('animation') animation = true;
  @Input('autofocus') autofocus = false;

  refill() {
    this.countries = countries;
  }
  filter($event: any) {
    this.expanded = true;
    if ($event.target.value.length > 0) {
      const transformation = this.titleCasePipe.transform(this.stringPipe.transform($event.target.value));
      this.countries = this.countries.filter(country => country.name.includes(transformation));
    }
  }

  pickFirst() {
    this.selectValue(this.countries[0]);
  }

  selectValue(option: any) {
    this.touched = true;
    this.selectedOption = option;

    if (this.control) {
      this.onTouch();
      this.onModelChange(option);
    } else this.valueEmitter.emit(option);
    this.expanded = false;
    this.inputRef?.nativeElement.blur();
  }

  @HostListener('document:click', ['$event'])
  documentClick(event: any) {
    if (!this.dropdownRef?.nativeElement?.contains(event.target)) {
      this.expanded = false;
      if (this.inputRef.nativeElement.value.length <= 0) {
        this.selectValue(this.selectedOption);
        this.inputRef.nativeElement.value = this.selectedOption.name;
      }
    }
  }

  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler() {
    this.expanded = false;
  }

  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {
    if (obj === null) {
      this.selectedOption = { code: "LB", name: "Lebanon" };
    }
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }
}
