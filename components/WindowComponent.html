import { DOCUMENT } from '@angular/common';
import { Component, EventEmitter, HostListener, Inject, Input, OnInit, Output } from '@angular/core';
import { faTimes } from '@fortawesome/free-solid-svg-icons';
import { dialogAnimation } from '../../../animations/dialog.animation';

@Component({
  selector: 'lib-window',
  templateUrl: './window.component.html',
  styleUrls: ['./window.component.scss'],
  animations: [dialogAnimation]
})
export class WindowComponent implements OnInit {

  constructor(@Inject(DOCUMENT) private document: Document) { }

  ngOnInit(): void {}
  
  faTimes = faTimes;

  @Input('title') title = "";
  @Input('closeIcon') closeIcon = true;
  @Input('large') large = false;

  @Input('visible') visible = false;
  @Output('visibleChange') visibleChange = new EventEmitter();

  @Output() onClose = new EventEmitter();

  open() {
    window.scrollTo(0,0);
    this.visible = true;
    this.document.getElementById('body')?.classList.add('scroll');
  }

  close() {
    this.visible = false;
    this.document.getElementById('body')?.classList.remove('scroll');
  }

  invokeClose() {
    this.visibleChange.emit();
    this.onClose.emit();
    this.close();
  }

  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler(event: KeyboardEvent) {
    this.close();
  }
}
