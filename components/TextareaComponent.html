import { Component, EventEmitter, Input, OnInit, Optional, Output, Self, ViewChild } from '@angular/core';
import { NgControl, ControlValueAccessor, Validators } from '@angular/forms';
import { CharPipe } from '../../../helpers/pipes/char.pipe';
import { MaxPipe } from '../../../helpers/pipes/max.pipe';

@Component({
  selector: 'lib-textarea',
  templateUrl: './textarea.component.html',
  styleUrls: ['./textarea.component.scss'],
  providers: [CharPipe, MaxPipe]
})
export class TextareaComponent implements OnInit, ControlValueAccessor {

  constructor(private charPipe: CharPipe, private maxPipe: MaxPipe,
    @Self() @Optional() public control: NgControl) {
      if (this.control) this.control.valueAccessor = this;
  }

  ngOnInit(): void {
  }

  @ViewChild('textareaRef') textareaRef: any;
  @Input('value') value = '';
  touched = false;

  private onTouch!: Function;
  private onModelChange!: Function;

  @Input('cols') cols: number = 5;
  @Input('rows') rows: number = 10;
  @Input('label') label: string | null = null;
  @Input('class') class: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('placeholder') placeholder = 'Type here ..';
  @Input('autofocus') autofocus = false;
  @Input('autocomplete') autocomplete = false;
  @Input('disabled') disabled = false;
  @Input('required') required = true;
  @Input('count') count: boolean = true;
  @Input('name') name: string | null = null;

  @Input('max') max: null | number = 5000;
  @Input('min') min = 50;
  @Input('isReactive') isReactive = true;
  @Output('onValueChange') valueEmitter = new EventEmitter();

  get invalid() {
    return this.isReactive ? this.control.touched && this.control.invalid && !this.disabled : 
    this.required && this.touched && this.textareaRef.nativeElement.value.length < this.min && !this.disabled;
  }

  ngAfterViewInit() {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      this.control.control?.addValidators(Validators.minLength(this.min))
    }
  }

  writeValue(obj: any): void {
    this.value = obj || '';
  }

  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }

  change() {
    this.touched = true;
    if (this.max !== null) {
      this.textareaRef.nativeElement.value =
      this.maxPipe
        .transform(this.value,
          this.max)
    }

    if (this.control) {
      this.onModelChange(this.value);
      this.onTouch();
    } else {
      this.valueEmitter.emit(this.value);
    }
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

}
