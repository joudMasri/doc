import { Component, OnInit, Output, EventEmitter, Input, ViewChild, Optional, Self } from '@angular/core';
import { ControlValueAccessor, NgControl } from '@angular/forms';
import { faFile } from '@fortawesome/free-regular-svg-icons';
import { faCloudUploadAlt, faPlus, faTimes } from '@fortawesome/free-solid-svg-icons';
import { MessageComponent } from './../../message/message/message.component';

@Component({
  selector: 'lib-file',
  templateUrl: './file.component.html',
  styleUrls: ['./file.component.scss']
})
export class FileComponent implements OnInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }

  onModelChange!: Function;
  onTouch!: Function;
  writeValue(obj: any): void {
    this.files = obj || null
  }

  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  ngOnInit(): void {
  }

  faTimes = faTimes;
  faPlus = faPlus;
  faUpload = faCloudUploadAlt;
  faFile = faFile;

  files: any[] = [];
  @Input('originalFiles') originalFiles: File[] = [];

  @Input('maxSize') maxSize: number = 1000000; // 1mb
  @Input('max') max: number = 3;
  @Input('multiple') multiple = true;
  @Input('basic') basic: boolean = false;
  @Input('accept') accept: string = '';
  @Input('name') name: string | null = null;
  @Input('required') required: boolean = true;
  @Input('disabled') disabled: boolean = false;
  @Input('chooseLabel') chooseLabel: string | null = null;
  @Input('chooseClass') chooseClass: string | null = null;
  @Input('uploadBtn') uploadBtn: boolean = true;
  @Input('uploadLabel') uploadLabel: string | null = null;
  @Input('uploadClass') uploadClass: string | null = null;
  @Input('autoUpload') autoUpload: boolean = false;
  @Output('onChange') onChange = new EventEmitter();
  @Output('onUpload') onUpload = new EventEmitter();
  @ViewChild('msgRef') msgRef!: MessageComponent;

  loading = false;

  get isValid() {
    return this.files.length > 0 && this.required || this.files.length > this.max;
  }

  addFile($event: any) {
    const file = ($event.target as any).files[0];
    const reader = new FileReader();
    reader.onload = () => {
      if (file.size < this.maxSize) {
        this.files.push({
          id: this.files.length,
          name: file.name,
          size: file.size,
          type: file.type
        })
        this.files[this.files.length - 1].output = (reader.result?.toString());
        this.originalFiles.push(file);
        if (this.control) {
          this.onModelChange(this.originalFiles);
          this.onTouch();
        }
        else {
          this.onChange.emit(this.originalFiles);
          if (this.autoUpload && this.basic) this.onUpload.emit(this.originalFiles);
        }
      }
      else {
        this.msgRef.open();
      }

    };

    reader.readAsDataURL(file);
  }


  cancelOne(id: number, name: string) {
    this.files.splice(id, 1);
    this.originalFiles = this.originalFiles.filter(f => f.name !== name);
    if (this.control) {
      this.onModelChange(this.originalFiles);
      this.onTouch();
    }
    else this.onChange.emit(this.originalFiles);
  }

  cancelAll() {
    this.files = [];
    this.originalFiles = [];
    if (this.control) {
      this.onModelChange(this.originalFiles);
      this.onTouch();
    }
    else this.onChange.emit(this.originalFiles);
  }

  upload() {
    this.onUpload.emit(this.originalFiles);
    this.loading = true;
  }
}
