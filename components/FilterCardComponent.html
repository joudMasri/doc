import { Component, Input, OnInit, Output, EventEmitter, ViewChild, ElementRef, HostListener, AfterViewInit } from '@angular/core';
import { SelectComponent } from './../../select/select/select.component';

@Component({
  selector: 'lib-filter-card',
  templateUrl: './filter-card.component.html',
  styleUrls: ['./filter-card.component.scss']
})
export class FilterCardComponent implements OnInit, AfterViewInit {

  constructor() { }

  ngOnInit(): void {
  }

  ngAfterViewInit() {
    if (this.type === 'number') {
      this.inputType = 'number';
      this.selectedOption = { label: 'Equals', value: 0 };
    }
  }

  empty = true;

  @Input('type') type: string = 'string';
  @Input('show') show: boolean = false;
  @Input('index') index: number;
  @Input('lg') lg = false;

  @Output('onApply') applyEmitter = new EventEmitter();
  @Output('onClear') clearEmitter = new EventEmitter();
  @Output('onExternalClick') externalClickEmitter = new EventEmitter();

  @ViewChild('inputRef') inputRef!: ElementRef<HTMLInputElement>;
  @ViewChild('elementRef') elementRef!: ElementRef<HTMLDivElement>;
  @ViewChild('selectRef') selectRef!: ElementRef<SelectComponent>;

  applied = false;
  onApply(option: any, target: any) {
    this.applied = true;
    this.applyEmitter.emit({ option: option, target: target });
    this.applied = true;
  }

  onClear() {
    this.applied = false;
    this.selectedOption = { label: 'Equals', value: 0 };
    if (this.inputRef) this.inputRef.nativeElement.value = '';
    this.clearEmitter.emit();
  }


  selectedOption = { label: 'Starts with', value: 0 };
  get options() {
    return this.type === 'number' ?
      this.numberOptions : this.stringOptions;
  }

  inputType: string = 'string';
  numberOptions = [
    { label: 'Equals', value: 0 },
    { label: 'Not equals', value: 1 },
    { label: 'Less than', value: 2 },
    { label: 'Less than or equal to', value: 3 },
    { label: 'Greater than', value: 4 },
    { label: 'Greater than or equal to', value: 5 }
  ];

  stringOptions = [
    { label: 'Starts with', value: 0 },
    { label: 'Contains', value: 1 },
    { label: 'Not contains', value: 2 },
    { label: 'Ends with', value: 3 },
    { label: 'Equals', value: 4 },
    { label: 'Not equals', value: 5 }
  ]

  @HostListener('document:click', ['$event'])
  close($event: any) {
    if (this.show) {
      this.externalClickEmitter.emit(this.index)
    }
    if (this.show && !this.elementRef.nativeElement.contains($event.target)
      && !$event.target?.classList.contains('showFilterCard')
      && !$event.target?.classList.contains('select-option')
      ) {
      this.show = false;
    }
  }

}
