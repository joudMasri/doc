import { Component, OnInit, ViewChild, AfterViewInit, Input, Optional, Self, Output, EventEmitter, ElementRef, ChangeDetectorRef } from '@angular/core';
import { ControlValueAccessor, NgControl, Validators } from '@angular/forms';
import { faEye, faEyeSlash, IconDefinition } from '@fortawesome/free-regular-svg-icons';
import { CharPipe } from '../../../helpers/pipes/char.pipe';
import { MaxPipe } from '../../../helpers/pipes/max.pipe';

@Component({
  selector: 'lib-reactive-input',
  templateUrl: './reactive-input.component.html',
  styleUrls: ['./reactive-input.component.scss'],
  providers: [CharPipe, MaxPipe],

})
export class ReactiveInputComponent implements OnInit, ControlValueAccessor, AfterViewInit {
  constructor(
    private charPipe: CharPipe,
    private maxPipe: MaxPipe,
    private cdRef: ChangeDetectorRef,
    @Self() @Optional() public control: NgControl) {
      if (this.control) this.control.valueAccessor = this;
    }

  ngOnInit(): void {
  }

  faEye = faEye;
  faEyeSlash = faEyeSlash;  

  @ViewChild('inputRef') inputRef!: ElementRef;

  focus = false;
  touched = false;

  get invalid() {
    if (this.isReactive) return this.control.touched && this.control.invalid && !this.disabled;
    else return this.touched && !this.disabled
    && ((this.required && this.type !== 'number' && this.currentValue.length < this.min)
    || (this.required && this.type === 'number' && this.currentValue.toString().length === 0)
    || (this.type === 'email' && !this.isEmailValid)
    || (this.type === 'password' && !this.isPatternValid))
  }

  get isValid() {
    return  ((this.required && this.type !== 'number' && this.currentValue.length < this.min)
    || (this.required && this.type === 'number' && this.currentValue.toString().length === 0)
    || (this.type === 'email' && !this.isEmailValid)
    || (this.type === 'password' && !this.isPatternValid))
  }

  get currentValue() {
    return this.inputRef?.nativeElement.value;
  }

  private onTouch!: Function;
  private onModelChange!: Function;

  @Input('label') label: string | null = null;
  @Input('type') type = 'text';
  @Input('class') class: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('placeholder') placeholder = 'Type here ..';
  @Input('autofocus') autofocus = false;
  @Input('autocomplete') autocomplete = false;
  @Input('disabled') disabled = false;
  @Input('changeColorOnDisable') changeColorOnDisable = true;
  @Input('isReactive') isReactive = true;
  @Input('required') required = true;
  @Input('sm') small = false;
  @Input('muted') muted : boolean = false;
  @Input('displayShowPasswordIcon') showPasswordIcon = true;
  @Input('iconRight') iconRight: null | IconDefinition = null;
  @Input('iconLeft') iconLeft: null | IconDefinition = null;
  @Input('digitAllowed') digitAllowed = true;
  @Input('letterAllowed') letterAllowed = this.type === 'number' ? false : true;
  @Input('specialAllowed') specialAllowed = this.type === 'number' ? false : true;
  @Input('spaceAllowed') spaceAllowed = this.type === 'number' ? false : true;
  @Input('validationMessage') validationMessage = true;
  @Input('max') max = 50;
  @Input('min') min = this.type === 'number' ? 0 : 3;
  @Input('name') name: string = '';
  @Input('loginPassword') loginPassword = false;
  @Output('onValueChange') valueEmitter = new EventEmitter();

  passwordRegEx = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)[a-zA-Z\d\w\W]{8,50}$/;
  emailRegEx = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/;

  ngAfterViewInit() {
    if (this.isReactive) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      if (this.type === 'password' && !this.loginPassword) {
        this.control.control?.addValidators(Validators.pattern(this.passwordRegEx));
      }
      if (this.type === 'email') {
        this.control.control?.addValidators(Validators.pattern(this.emailRegEx))
      }
      if (this.type === 'number') {
        this.control.control?.addValidators(Validators.pattern(/^[0-9]*$/))
      }
      if (this.type === 'text') {
        this.control.control?.addValidators([Validators.minLength(this.min), Validators.maxLength(this.max)])
      }
    }
    this.cdRef.detectChanges();
  }

  @Input('value') value: any = '';
  writeValue(obj: any): void {
    this.value = obj || '';
  }

  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }

  set currentValue(value: any) {
    if (this.inputRef) this.inputRef.nativeElement.value = value;
  }

  change($event: any) {
    this.touched = true;
    const eventValue = $event.target.value;
    this.currentValue =
    this.charPipe
      .transform(eventValue,
        this.digitAllowed,
        this.letterAllowed,
        this.specialAllowed,
        this.spaceAllowed
      );

    if (this.max !== null && this.type !== 'number' && this.type !== 'password') {
      this.currentValue =
      this.maxPipe
        .transform(this.currentValue,
          this.max)
    }

    if (this.type === 'password') {
      this.currentValue = this.maxPipe.transform(this.currentValue, 16);
    }
    if (this.type === 'number') {
      this.inputRef.nativeElement.valueAsNumber > this.max ? this.currentValue = this.max : null;
      this.inputRef.nativeElement.valueAsNumber < this.min ? this.currentValue = this.min : null;
    }

    if (this.control) {
      this.onModelChange(this.currentValue);
      this.onTouch();
    } else {
      this.valueEmitter.emit(this.currentValue);
    }
  }

  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  get isPatternValid() : boolean {
    return this.currentValue?.match(this.passwordRegEx);
  }

  get isEmailValid() {
    return this.currentValue?.match(this.emailRegEx);
  }

}

      </div>

        * Please enter a valid email.
      </div>

        * Password contains between 8 and 16 characters, one uppercase and one number.
      </div>

  </small>
