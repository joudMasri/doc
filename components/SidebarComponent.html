import { trigger, transition, useAnimation, animate, keyframes, style } from '@angular/animations';
import { DOCUMENT } from '@angular/common';
import { Component, EventEmitter, HostListener, Inject, Input, OnInit, Output, ViewChild } from '@angular/core';
import { faTimes } from '@fortawesome/free-solid-svg-icons';
import { sidebarAnimations } from '../../../animations/sidebar.animation';

@Component({
  selector: 'lib-sidebar',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss'],
  animations: [sidebarAnimations]
})
export class SidebarComponent implements OnInit {

  constructor(@Inject(DOCUMENT) private document: Document) { }

  ngOnInit(): void {
  }

  faTimes = faTimes;

  @ViewChild('sidebarRef') sidebarRef: any;
  @ViewChild('maskRef') maskRef: any;

  @Input('modal') modal = true;
  @Input('title') title = '';
  @Input('dismissible') dismissible = true;
  @Input('full') full = false;
  @Input('class') class: string | null = null;
  @Input('right') right: boolean = false;

  @Input('visible') visible = false;
  @Output('visibleChange') visibleChange = new EventEmitter();

  buttonRef: any;

  open(ref: any) {
    this.visible = true;
    this.visibleChange.emit();
    this.buttonRef = ref;
    this.document.getElementById('body')?.classList.add('scroll');
  }

  close() {
    this.visible = false;
    this.visibleChange.emit();
    this.document.getElementById('body')?.classList.remove('scroll');
  }

  @HostListener('document:click', ['$event'])
  documentClick(event: any) {
    if (this.dismissible && this.visible) {
      if (this.modal
        && this.maskRef?.nativeElement.contains(event.target)) {
        this.close();
      }

      if (!this.modal
          && !this.sidebarRef?.nativeElement.contains(event.target)
          && !this.buttonRef?.contains(event.target)) {
        this.close();
      }
    }
  }

  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler(event: KeyboardEvent) {
    this.close();
  }

}
