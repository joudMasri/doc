import { AfterViewInit, ChangeDetectorRef, Component, ElementRef, EventEmitter, HostListener, Input, OnInit, Optional, Output, Self, ViewChild } from '@angular/core';
import { ControlValueAccessor, NgControl, Validators } from '@angular/forms';
import { faCalendarAlt } from '@fortawesome/free-regular-svg-icons';
import { faChevronLeft, faChevronRight } from '@fortawesome/free-solid-svg-icons';
import { DateService } from '../../../services/date.service';
import { swingIn } from '../../../animations/calendar.animation';
import { fade } from '../../../animations/shared.animation';
import { Day } from '../../../models/events';

@Component({
  selector: 'lib-date-picker',
  templateUrl: './date-picker.component.html',
  styleUrls: ['./date-picker.component.scss'],
  animations: [swingIn, fade]
})
export class DatePickerComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl, public dateService: DateService, 
  private cdRef: ChangeDetectorRef) {
    if (this.control) this.control.valueAccessor = this;
  }

  faChevronLeft = faChevronLeft;
  faChevronRight = faChevronRight;
  faCalendar = faCalendarAlt;

  ngOnInit(): void {
    this.reset();
  }

  selectedDay!: number;
  selectedMonth!: number;
  selectedYear!: number;
  selectedMonthDays: Day[] = [];
  years: number[] = [];
  view = 'days';
  show = false;
  focused = false;

  @ViewChild('inputRef') inputRef!: ElementRef<HTMLInputElement>;
  @ViewChild('calendarRef') calendarRef!: ElementRef;

  @Input('inline') inline = false;
  @Input('static') static = false;
  @Input('required') required = true;
  @Input('autofocus') autofocus = false;
  @Input('name') name: string | null = null;
  @Input('class') class: string | null = null;
  @Input('label') label = '';
  @Input('labelTop') labelTop = false;
  @Input('dateRange') dateRange = false;
  @Input('disabled') disabled = false;
  @Input('maxDate') maxDate : { year: number, month: number, day: number } = { year: 3000, month: 0, day: 1};
  @Input('minDate') minDate : { year: number, month: number, day: number } = { year: 1990, month: 0, day: 1};
  @Output('onValueChange') valueEmitter = new EventEmitter();
  @Input('sm') small = false;
  @Input('isReactive') isReactive = true;

  toggle() {
    this.show = !this.show;
    this.view = 'days';
  }

  startDate: { day?: number, month?: number, year?: number } = {};
  endDate: { day?: number, month?: number, year?: number } = {};

  get invalid() {
    return this.isReactive ? this.control.touched && this.control.invalid && !this.disabled
    : this.touched && this.inputRef.nativeElement.value === 'dd/mm/yyyy' && !this.disabled;
  }

  selectDay(day: number) {
    this.selectedDay = day;
    this.fallbackDate();
    if (!this.dateRange) {
      this.show = false;
    }
    else if (this.dateRange) {
      this.setDateRange(day)
    }
    this.formatInputValue();
    this.emitDate();

  }

  selectMonth(month: number) {
    if (!(this.selectedYear >= this.maxDate.year && month > this.maxDate.month)
    || (this.selectedYear <= this.minDate.year && month < this.minDate.month)) {
      this.selectedMonth = month;
      this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, []);
      this.fallbackDate();
      this.formatInputValue();
      this.view = 'days';
      this.emitDate();
    }
  }

  selectYear(year: number) {
    if (!(year > this.maxDate.year || year < this.minDate.year)) {
      this.selectedYear = year;
      this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, []);
      this.fallbackDate();
      this.formatInputValue();
      this.view = 'days';
      this.emitDate();
    }
  }

  touched = false;

  emitDate() {
    this.touched = true;
    let value: any;
    let endValue: any;
    if (this.dateRange) {
      value =
      { startDate: this.startDate.day ? this.startDate
        : { day: this.selectedDay, month: this.selectedMonth, year: this.selectedYear },
        endDate: this.endDate.day ? this.endDate : this.startDate.day? this.startDate :
        { day: this.selectedDay, month: this.selectedMonth, year: this.selectedYear }
      }

      endValue = {
        startDate: this.fromObjectToDateString(value.startDate),
        endDate: this.fromObjectToDateString(value.endDate)
      }
    }
    else endValue = new Date(this.selectedYear, this.selectedMonth, this.selectedDay, 0, 0, 0);

    if (this.control) {
      this.onTouch();
      this.onModelChange(endValue);
    } else this.valueEmitter.emit(endValue);

  }

  ngAfterViewInit() {
    this.startDate = this.obj?.startDate;
    this.endDate = this.obj?.endDate;
    this.formatInputValue();
    if (!this.obj && !this.dateRange) {
      this.inputRef.nativeElement.value = `dd/mm/yyyy`;
    }
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
    }
  }

  increment() {
    if (!(this.selectedYear >= this.maxDate.year
        && this.selectedMonth >= this.maxDate.month)) {
      if (this.selectedMonth === 11) {
        this.selectedMonth = 0;
        this.selectedYear++;
      } else this.selectedMonth = this.selectedMonth + 1;
      this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, []);
    }
  }

  decrement() {
    if (!(this.selectedYear <= this.minDate.year
      && this.selectedMonth <= this.minDate.month)) {
        if (this.selectedMonth === 0) {
          this.selectedMonth = 11;
          this.selectedYear--;
        } else this.selectedMonth = this.selectedMonth - 1;
        this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, []);
      }
  }

  getYears(year: number) {
    if (year <= this.maxDate.year && year >= this.minDate.year) {
      this.years = [];
      let startingYear = +(year.toString().substring(0,3) + '0');
      let endYear = startingYear + 10;
      for (let i = startingYear; i <= endYear; i++) {
        this.years.push(i)
      }
    }
  }

  fromObjectToDateString(object: {day?: number, month?: number, year?: number}) {
    return new Date(Date.UTC(object?.year || 0, object?.month || 0, object?.day || 0, 0, 0, 0, 0))
  }

  getInRange(day: number) {
    return this.dateService
    .getInRange(day, this.selectedMonth, this.selectedYear,
      this.fromObjectToDateString(this.startDate), this.fromObjectToDateString(this.endDate))
  }

  getRangeEnd(day: number) {
    return this.dateService
      .getRangeEnd(day, this.selectedMonth, this.selectedYear,
        this.fromObjectToDateString(this.startDate), this.fromObjectToDateString(this.endDate))
  }

  setDateRange(day: number) {
    if (this.startDate && this.endDate && this.startDate.day === this.endDate.day
      && this.startDate.month === this.endDate.month
      && this.startDate.year === this.endDate.year) {
        if (( this.startDate.day
          && day < this.startDate.day
          && this.startDate.month
          && this.startDate.month >= this.selectedMonth
          && this.startDate.year
          && this.startDate.year >= this.selectedYear)
          || (this.startDate.month
            && this.startDate.month > this.selectedMonth
            && this.startDate.year
            && this.startDate.year >= this.selectedYear
          )
        ) {
          this.setStartDate(day);
        } else {
          this.endDate.day = day;
          this.endDate.month = this.selectedMonth;
          this.endDate.year = this.selectedYear;
        }
    }
    else if (!this.startDate.day) {
      this.setStartDate(day);
    } else if (!this.endDate.day) {
      if ((day < this.startDate.day
        && this.startDate.month
        && this.startDate.month >= this.selectedMonth
        && this.startDate.year
        && this.startDate.year >= this.selectedYear)
        || (this.startDate.month
          && this.startDate.month > this.selectedMonth
          && this.startDate.year
          && this.startDate.year >= this.selectedYear
        )
      ) {
        this.setStartDate(day);
      } else {
        this.endDate.day = day;
        this.endDate.month = this.selectedMonth;
        this.endDate.year = this.selectedYear;
      }
    } else {
      this.startDate = {};
      this.endDate = {};
      this.setStartDate(day);
    }
  }

  setStartDate(day: number) {
    this.startDate.day = day;
    this.startDate.month = this.selectedMonth;
    this.startDate.year = this.selectedYear;
  }

  reset() {
    let date = new Date();
    this.selectedDay = this.obj ? this.dateService.fromDateStringToObject(this.obj).day : date.getDate();
    this.selectedMonth = this.obj ? this.dateService.fromDateStringToObject(this.obj).month : date.getMonth();
    this.selectedYear =  this.obj ? this.dateService.fromDateStringToObject(this.obj).year : date.getFullYear();
    this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, []);
    this.getYears(this.selectedYear);

    if (this.dateRange) {
      this.endDate = {};
      this.startDate = {};
    }
  }

  obj: any;

  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {
    if (this.obj === null) {
      this.reset();
      this.ngAfterViewInit();
    }
    this.obj = obj;
    this.cdRef.detectChanges();
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  @HostListener('document:click', ['$event'])
  documentClick(event: any) {
    if (!this.calendarRef?.nativeElement?.contains(event.target)) {
      this.show = false;
      this.view = 'days';
    }
  }

  @HostListener('document:keydown.escape', ['$event'])
  onKeydownHandler(event: KeyboardEvent) {
    this.show = false;
    this.view = 'days';
  }

  transformMonths(month: number) : string {
    let m = month + 1;
    return m.toString().length === 1 ? `0${m}` : m.toString();
  }

  private formatInputValue() : void {
    let year = this.selectedYear;
    let month = this.transformMonths(this.selectedMonth);
    let day = this.selectedDay.toString().length === 1 ? `0${this.selectedDay}` : this.selectedDay;
    if (!this.dateRange) this.inputRef.nativeElement.value = `${day}/${month}/${year}`;
    else {
      let sm = this.startDate.month ? this.transformMonths(this.startDate.month) : null;
      let em = this.endDate.month ? this.transformMonths(this.endDate.month) : null;
      if (!this.startDate.day) {
        this.inputRef.nativeElement.value = `${day}/${month}/${year}` + ` - ${day}/${month}/${year}`;
      } else if (this.endDate.day) {
        this.inputRef.nativeElement.value =
        `${this.startDate.day}/${sm}/${this.startDate.year}`
        + ` - ${this.endDate.day}/${em}/${this.endDate.year}`;
      } else {
        this.inputRef.nativeElement.value = `${this.startDate.day}/${sm}/${this.startDate.year}`
        + ` - ${sm}/${this.startDate.day}/${this.startDate.year}`;
      }
    }
  }

  private fallbackDate() {
    if (this.selectedDay > this.dateService.getDaysCount(this.selectedMonth, this.selectedYear)) {
      this.selectedDay = this.dateService.getDaysCount(this.selectedMonth, this.selectedYear)
    }

    let minDate = this.fromObjectToDateString(this.minDate);
    let maxDate = this.fromObjectToDateString(this.maxDate);
    let currentDate = 
    this.fromObjectToDateString({day: this.selectedDay, month: this.selectedMonth, year: this.selectedYear})
    if (currentDate > maxDate) {
      this.selectedDay = this.maxDate.day;
      this.selectedMonth = this.maxDate.month;
      this.selectedYear = this.maxDate.year;
    }

    if (currentDate < minDate) {
      this.selectedDay = this.minDate.day;
      this.selectedMonth = this.minDate.month;
      this.selectedYear = this.minDate.year;
    }
  }

}
