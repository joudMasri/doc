import { Component, Input, OnInit, Output, EventEmitter,  Optional, Self, AfterViewInit } from '@angular/core';
import { NgControl, ControlValueAccessor, Validators } from '@angular/forms';
import { fade } from '../../../animations/shared.animation';


@Component({
  selector: 'lib-chip',
  templateUrl: './chip.component.html',
  styleUrls: ['./chip.component.scss'],
  animations: [fade]
})
export class ChipComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }
  ngAfterViewInit(): void {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      this.control.control?.addValidators([Validators.minLength(this.min)])
    }
  }

  onTouch!: Function;
  onModelChange!: Function;

  writeValue(obj: any): void {
    if (typeof(obj) !== 'string')
      this.selectedChips = obj || [];
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }

  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  ngOnInit(): void {
    if (!this.max) {
      this.max = this.chips.length;
    }
  }

  @Input('chips') chips: any[] = [];
  @Input('class') class: string | null = null;
  @Input('label') label: string | null = null;
  @Input('required') required = true;
  @Input('min') min = 0;
  @Input('max') max: number | null = null;
  @Input('isReactive') isReactive = true;

  @Output('onValueChange') onValueChange = new EventEmitter();

  touched = false;
  onClick(chip: string) {
    if (this.control) this.onTouch();
    this.touched = true;

    if (this.selectedChips.includes(chip)) {
      this.selectedChips = this.selectedChips.filter(c => c !== chip);
      if (this.control) this.onModelChange(this.selectedChips);
    }
    else if (this.max !== null && this.selectedChips.length === this.max) {
      return;
    }
    else {
      this.selectedChips.push(chip);
      if (this.control) this.onModelChange(this.selectedChips);
    }

    this.onValueChange.emit(this.selectedChips);
  }

  selectedChips: string[] = [];
}
