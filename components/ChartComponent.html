import { AfterViewInit, ChangeDetectorRef, Component, ElementRef, HostListener, Input, OnChanges, OnInit, SimpleChanges, ViewChild } from '@angular/core';
export class Chart {
  hStep!: number;
  wStep!: number;
}
@Component({
  selector: 'lib-chart',
  templateUrl: './chart.component.html',
  styleUrls: ['./chart.component.scss']
})
export class ChartComponent implements OnInit, OnChanges, AfterViewInit {

  constructor(private cdRef: ChangeDetectorRef) { }

  @ViewChild('svg') svg!: ElementRef;
  @ViewChild('overlay') overlay!: ElementRef<HTMLDivElement>;
  @ViewChild('donutCircle') donutCircle!: ElementRef<SVGCircleElement>;

  
  ngAfterViewInit(): void {
    if (this.type === 'pie') {
      this.svg.nativeElement.style.width = '100%';
      this.svg.nativeElement.style.height = `${this.pw}px`;
    }
  }

  ngOnChanges(changes: SimpleChanges): void {
  
    if (changes['data']?.currentValue) {
      this.data = changes['data']?.currentValue;
      this.xValues = changes['data']?.currentValue.labels;
      this.yValues = changes['yValues']?.currentValue;
    }

    if (changes['type'].currentValue === 'pie' && changes['pieData']?.currentValue.length === 0 && changes['data']?.currentValue.datasets.length > 1) {
      throw new Error(`Please make sure you entered valid pieData; e.g: [{label: .., value: .., backgroundColor?: ..}, ..]
      warning: Do not use pie chart when you have more than one dataset`);
    } 
    
    if (changes['type'].currentValue === 'pie' && this.data.datasets.length === 1) {
      this.pieData = [];
      this.data.labels.forEach((element, index) => {
        this.pieData.push({
          label: element,
          data: this.data.datasets[0].data[index],
        })
      });
      this.total = this.pieData.reduce((acc, obj) => acc + obj.data, 0);
      this.ngAfterContentInit();
    }

    if (changes['type']?.currentValue === 'radar') {
      this.rads = [];
      this.yValues.forEach((element: number) => {
        this.rads.push(element/2 * 10)
      });
      this.setUp();
      this.radarViewBox = `-${this.yValues[this.yValues.length - 1]*7/2*2.8} -${this.yValues[this.yValues.length - 1]*7/2*2.8} ${this.yValues[this.yValues.length - 1]*7*2.8} ${this.yValues[this.yValues.length - 1]*7*2.8}`
    }
    this.chart.hStep = changes['yValues']?.currentValue.length ? changes['yValues']?.currentValue.length 
    : this.yValues.length * this.yf;
    
    this.chart.wStep = changes['type'].currentValue === 'line' || changes['type'].currentValue === 'point'  ? 
    (this.xValues.length - 1) * this.xf : changes['type'].currentValue === 'horizontal' ? 
    this.yValues.length * this.xf : this.xValues.length * this.xf;
    
    this.chart.hStep = changes['type'].currentValue === 'horizontal' ? this.xValues.length * this.yf 
    : this.yValues.length * this.yf;

    this.chart.wStep = changes['type'].currentValue === 'line' || changes['type'].currentValue === 'point' ? 
    (this.xValues.length - 1) * this.xf : changes['type'].currentValue === 'horizontal' ? 
    this.yValues.length * this.xf : this.xValues.length * this.xf;

    this.h = this.chart.hStep + this.margin * 2;
    this.w = this.chart.wStep + this.margin * 2;
 
    this.viewBox = changes['type'].currentValue === 'pie' ? 
    `${-(this.pw/40)} ${-(this.pw/40)} ${this.pw/20} ${this.pw/20}` : '0 0 ' + this.w + ' ' + this.h; 
    this.tl = ` ${this.margin} ${this.margin} `;
    this.tr = ` ${this.w - this.margin} ${this.margin} `;
    this.bl = ` ${this.margin} ${this.h - this.margin} `;
    this.br = ` ${this.w - this.margin} ${this.h - this.margin} `;
    this.tlx = this.margin;
    this.tly = this.margin;
    this.trx = this.w - this.margin;
    this.try = this.margin;
    this.blx = this.margin;
    this.bly = this.h - this.margin;
    this.brx = this.w - this.margin;
    this.bry = this.h - this.margin;


  }
 
  ngOnInit(): void {
    this.type === 'radar' ? this.setUp() : null;
  }

  interval: any;
  ngAfterContentInit(): void {
    this.transformValuesToPercent();
    this.interval = setInterval(() => this.increment(), this.pieData.length * 2);
  }

  increment() {
    let i = 0;
    this.percent -= this.pieData[i].data/this.total;
    if (this.percent <= 0) {
      this.percent = 0;
      clearInterval(this.interval);
    }
    i++;
  }

  getPathD(numbers: number[]) {
    let points: string = 'M ';
    for (let i = 0; i < numbers.length; i++) {
      points +=(`${this.blx + i * this.xf},${(this.yValues.length) + 1 - numbers[i]/10}`);
      if (!(i === numbers.length - 1)) {
        points += ' L '
      }
    }
    return points;
  }

  getBefore(numbers: number[]) {
    let points: string = 'M ';
      for (let i = 0; i < numbers.length; i++) {
        points +=(` ${this.blx + i * this.xf},${this.h - this.margin} L`);
      }
    return points;
  }
  get step() {
    return this.chart.wStep/(this.data.labels.length);
  };

  get strokeWidth() {return this.xf/(1.5 * this.data.datasets.length);}

  getBarPath(i: number, j: number) {
    return `M ${this.margin + this.step/2 + (this.step * i) 
    + this.getSpace(j, this.strokeWidth)} ${this.bly} V ${(this.yValues.length + 1) - (this.data.datasets[j].data[i])/10}`
  }

  getBarBefore(i: number, j: number) {
    return `M ${this.margin + this.step/2 + (this.step * i) 
    + this.getSpace(j, this.strokeWidth)} ${this.bly} V ${this.bly}`
  }

  get hoStep() {
    return this.chart.hStep/(this.data.labels.length);
  }

  get strokeWidthForH() {return this.yf/(2 * this.data.datasets.length)}
  
  getSpace(j: number, strokeWidth: number) {
    if (this.data.datasets.length % 2 === 0) {
      let compared = this.data.datasets.length/2;
      if (j < compared) {
        return -((compared - j) * strokeWidth - strokeWidth/2);
      } else if (j === compared) {
        return strokeWidth/2;
      } else {
        return (j - compared) * strokeWidth  + strokeWidth/2;
      }
    } else {
      let compared = Math.floor(this.data.datasets.length/2);
      if (j < compared) {
        return (-j - 1) * strokeWidth;
      } else if (j === compared) {
        return 0;
      } else {
        return ((j) - compared) * strokeWidth;
      }
    }
  } 

  getHorizontalPath(i: number, j: number) {
    return `M ${this.margin} ${this.margin + this.hoStep/2 + (this.hoStep * i) 
    + this.getSpace(j, this.strokeWidthForH)} H ${
      (this.data.datasets[j].data[i]) * (this.brx - this.margin) / (this.yValues[this.yValues.length - 1] * 10) + this.margin
    }`
  }
 
  getHorizontalBefore(i: number, j: number) {
    return `M ${this.margin} ${this.margin + this.hoStep/2 + (this.hoStep * i) 
    + this.getSpace(j, this.strokeWidthForH)} H ${
      this.blx
    }`
  }

  // legend
  lw = this.data.datasets.length + this.margin * 2;
  lh = this.lw/25;
  c = this.lw/100;


  // Pie
  total = this.pieData.reduce((acc, obj) => acc + obj.data, 0);
  percent = 1;
  pw = 500;
  radius = 50;
  squareHeight = 1;

  transformValuesToPercent() {
    this.pieData = this.pieData.sort((a,b) => a.data - b.data);
    let cumulativePercent = 0;

    this.pieData.forEach(element => {
      element.percent = element.data/this.total;
      element.startX = this.getCoordinatesForPercent(cumulativePercent)[0];
      element.startY = this.getCoordinatesForPercent(cumulativePercent)[1]

      cumulativePercent += element.percent;

      element.endX = this.getCoordinatesForPercent(cumulativePercent)[0];
      element.endY = this.getCoordinatesForPercent(cumulativePercent)[1];
    });

  }

  getCoordinatesForPercent(percent: number, radius: number = this.radius) {
    const x = Math.cos(2 * Math.PI * percent) * radius;
    const y = Math.sin(2 * Math.PI * percent) * radius;
    return [x, y];
  }

  getPoints(radius: number) {
    let cumulativePercent = 0.75;
    let percent = 1/this.data.labels.length;
    for (let i = 0; i < this.data.labels.length; i++) {
      let el: any = {};
      el.startX = this.getCoordinatesForPercent(cumulativePercent, radius)[0];
      el.startY = this.getCoordinatesForPercent(cumulativePercent, radius)[1];

      cumulativePercent += percent;

      el.endX = this.getCoordinatesForPercent(cumulativePercent, radius)[0];
      el.endY = this.getCoordinatesForPercent(cumulativePercent, radius)[1];
      this.els.push(el);
    }

  }

    let arr = [...this.els];

  }


  getLines(slice: any) {
    let lines: {x1?: number, y1?: number, x2?: number, y2?: number}[] = [];
    for (let i = 0; i < slice.length; i++) {
      lines.push({x1: slice[i].startX, y1: slice[i].startY, x2: slice[i].endX, y2: slice[i].endY})
    }
    return lines;
  }

  getValuesPoints(numbers: number[]) {
    let lines: string = 'M ';
    let cumulativePercent = 0.75;
    let percent = 1/this.data.labels.length;

    for (let i = 0; i < numbers.length; i++) {
      let el: any = {};
      el.startX = this.getCoordinatesForPercent(cumulativePercent, numbers[i]/2)[0];
      el.startY = this.getCoordinatesForPercent(cumulativePercent, numbers[i]/2)[1];

      cumulativePercent += percent;

      el.endX = this.getCoordinatesForPercent(cumulativePercent, numbers[i]/2)[0];
      el.endY = this.getCoordinatesForPercent(cumulativePercent, numbers[i]/2)[1];
      lines += `${el.startX},${el.startY} `;
      if (!(i === numbers.length - 1)) {
        lines += ' L '
      }
      if (i === numbers.length - 1) {
        lines += 
        `${this.getCoordinatesForPercent(cumulativePercent, numbers[0]/2)[0]},${this.getCoordinatesForPercent(cumulativePercent, numbers[0]/2)[1]} `;
      }
    }

    return lines;
  }
