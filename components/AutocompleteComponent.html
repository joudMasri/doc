import { Component, ElementRef, EventEmitter, HostListener, Input, OnInit,
  Optional, Output, Self, ViewChild, ViewChildren, QueryList, AfterViewInit } from '@angular/core';
import { ControlValueAccessor, NgControl, Validators } from '@angular/forms';
import { CharPipe } from '../../../helpers/pipes/char.pipe';
import { MaxPipe } from '../../../helpers/pipes/max.pipe';

@Component({
  selector: 'lib-autocomplete',
  templateUrl: './autocomplete.component.html',
  styleUrls: ['./autocomplete.component.scss'],
  providers: [CharPipe, MaxPipe]
})
export class AutocompleteComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl, private charPipe: CharPipe, private maxPipe: MaxPipe) {
    if (this.control) this.control.valueAccessor = this;
  }
  
  ngAfterViewInit(): void {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }

      this.control.control?.addValidators(Validators.minLength(this.min))
    }
  }

  onTouch!: Function;
  onModelChange!: Function;
  writeValue(obj: any): void {
    this.selectedValue = obj || null;
  }
  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  ngOnInit(): void {}

  touched = false;

  get invalid() {
    return (this.control?.touched || this.touched)
    && this.required && (this.control?.invalid || this.inputRef.nativeElement.value.length < this.min)
  }

  @ViewChild('inputRef') inputRef!: ElementRef;
  @ViewChildren('listRef') listRef!: QueryList<ElementRef>;

  @Input('items') items: string[] = [];
  @Input('selectedValue') selectedValue: string | null = null;
  @Input('label') label: string | null = null;
  @Input('class') class: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('placeholder') placeholder = 'Type here ..';
  @Input('autofocus') autofocus = false;
  @Input('autocomplete') autocomplete = false;
  @Input('disabled') disabled = false;
  @Input('required') required = true;
  @Input('name') name: string | null = null;

  @Input('digitAllowed') digitAllowed = true;
  @Input('letterAllowed') letterAllowed = true;
  @Input('specialAllowed') specialAllowed = true;
  @Input('spaceAllowed') spaceAllowed = true;

  @Input('max') max: null | number = null;
  @Input('min') min = 3;

  @Input('sm') small = false;

  @Output('onSelect') onSelect = new EventEmitter();

  results: string[] = [];
  focus = false;

  filter($event: any) {
    this.change($event)
    if ($event.target.value !== null && $event.target.value.length !== '') {
      this.results = this.items
      .filter(i => i.toLowerCase()
        .includes($event.target.value
          .toLowerCase()
        )
      )
    } else {
      this.results = [];
    }

    if (this.control) this.onTouch();
  }

  selectValue(value: string) {
    this.selectedValue = value;
    if (this.control) this.onModelChange(this.selectedValue);
    else this.onSelect.emit(this.selectedValue);
    this.results = [];

  }

  fill($event: any) {
    if (this.selectedValue === null || this.selectedValue === '')
      this.results = [];
    else this.filter($event);
  }

  empty() {
    this.results = [];
    this.selectedValue = null;
    if (this.control) this.onModelChange(this.selectedValue);
    else this.onSelect.emit(this.selectedValue);
  }

  onBlur() {
    this.focus = false;
    if (this.control) this.onModelChange(this.inputRef.nativeElement.value);
    else this.onSelect.emit(this.inputRef.nativeElement.value);
  }

  change($event: any) {
    const eventValue = $event.target.value;
    this.inputRef.nativeElement.value =
    this.charPipe
      .transform(eventValue,
        this.digitAllowed,
        this.letterAllowed,
        this.specialAllowed,
        this.spaceAllowed
      );

    if (this.max !== null) {
      this.inputRef.nativeElement.value =
      this.maxPipe
        .transform(this.inputRef.nativeElement.value,
          this.max)
    }

    if (this.control) {
      this.onModelChange(this.inputRef.nativeElement.value);
      this.onTouch();
    }
  }

  @HostListener('document:click', ['$event'])
  onExternalClick($event: any) {
    if (!$event.target.classList.contains('select-option')
    && !this.inputRef.nativeElement.contains($event.target)) {
      this.results = []
    }
  }
}

  <div

    [class.focused]="focus"
    [class.invalid]="invalid"
    >
    <input #inputRef
      class="muted-input"
      type="text"
      placeholder="Search .."
      (input)="filter($event); touched = true"
      (focus)="focus = true; results = items"
      (keydown.enter)="filter($event)"
      (keyup.backspace)="fill($event)"
      (blur)="onBlur()"
      [value]="selectedValue"
      [placeholder]="placeholder"
      [required]="required"
      [disabled]="disabled"
      [autofocus]="autofocus"
      [autocomplete]="autocomplete"
      [name]="name"
    >
    <span *ngIf="label && !labelTop" class="float-label f-sm">{{ label | titlecase}}
      <span *ngIf="required" class="color-error">*</span>
    </span>
    <div class="absolute dropdown-list" *ngIf="results.length > 0" style="z-index: 5000">
      <div class="select-option"
        #listRef
        *ngFor="let result of results"
        tabindex="0"
        (keydown.enter)="selectValue(result)"
        (click)="selectValue(result)"
        >
        {{ result }}
      </div>
    </div>
  </div>
  <small class="f-sm color-error" *ngIf="control?.touched || touched">
    <div *ngIf="control.hasError('required') || (required && inputRef.value.length === 0)">
      * This field is required.
    </div>
    <div *ngIf="inputRef.value.length > 0 && (control.hasError('minlength') || (inputRef.value.length < min))">
      * Type at least {{ min }} characters.
    </div>
  </small>
  <ng-content></ng-content>
</div>
