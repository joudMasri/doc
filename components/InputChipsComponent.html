import { Component, Input, OnInit, Optional, Output, Self, ViewChild, EventEmitter, HostListener, AfterViewInit } from '@angular/core';
import { NgControl, ControlValueAccessor, Validators } from '@angular/forms';
import { faExclamation, faTimes } from '@fortawesome/free-solid-svg-icons';
import { fade } from '../../../animations/shared.animation';

@Component({
  selector: 'lib-input-chips',
  templateUrl: './input-chips.component.html',
  styleUrls: ['./input-chips.component.scss'],
  animations: [fade]
})
export class InputChipsComponent implements OnInit, AfterViewInit, ControlValueAccessor {

  constructor(@Self() @Optional() public control: NgControl) {
    if (this.control) this.control.valueAccessor = this;
  }

  private onTouch!: Function;
  private onModelChange!: Function;
  writeValue(obj: any): void {
    this.chips = obj || [];
  }

  registerOnChange(fn: any): void {
    this.onModelChange = fn;
  }
  registerOnTouched(fn: any): void {
    this.onTouch = fn;
  }

  ngOnInit(): void {}

  ngAfterViewInit(): void {
    if (this.control) {
      if (this.required) {
        this.control.control?.addValidators(Validators.required)
      }
      this.control.control?.addValidators(Validators.minLength(this.min))
    }
  }


  faTimes = faTimes;
  faExclamation = faExclamation;

  chips: {label: string, value: any}[] = [];
  duplication = false;
  touched = false;
  focused = false;

  @ViewChild('inputRef') inputRef: any;

  @Input('placeholder') placeholder = 'Press enter to add value ..';
  @Input('label') label : string | null = null;
  @Input('class') class: string | null = null;
  @Input('autofocus') autofocus = false;
  @Input('autocomplete') autocomplete = false;
  @Input('name') name: string | null = null;
  @Input('labelTop') labelTop = false;
  @Input('required') required = true;
  @Input('min') min = 1;
  @Input('allowDuplication') allowDuplication = false;
  @Input('spaceAllowed') spaceAllowed = false;
  @Input('max') max = 15;
  @Input('sm') small = false;
  @Input('disabled') disabled = false;
  @Input('suggestions') suggestions: string[] = ['one', 'two', 'three'];
  @Output('onValueChange') onValueChange = new EventEmitter();
  @Input('isReactive') isReactive = true;

  get invalid() {
    return this.isReactive ? this.control.touched && (this.control.invalid || this.duplication) && !this.disabled : 
    (this.touched && this.chips.length < this.min || this.duplication) && !this.disabled
  }

  addChip($event: any, suggestedResult?: boolean) {
    this.touched = true;
    const content = suggestedResult ? $event : $event.target.value.trimStart();
    if (content.length > 0 && this.chips.length < this.max) {
      if (!this.allowDuplication
          && this.isDuplicate(content)
      ) {
        this.duplication = true;
      } else {
        this.chips.push({label: content, value: content});
        this.inputRef.nativeElement.value = null;
        this.results = [];
      }

      if (this.control) this.onModelChange(this.chips);
      else this.onValueChange.emit(this.chips);
    }

    if (this.control) this.onTouch();
  }

  remove(index: number) {
    this.chips.splice(index, 1);
    if (this.control) this.onModelChange(this.chips);
    else this.onValueChange.emit(this.chips);
  }

  pop() {
    if (this.inputRef.nativeElement.value === null || this.inputRef.nativeElement.value === '')
      this.chips.pop();
    if (this.control) this.onModelChange(this.chips);
    else this.onValueChange.emit(this.chips);
  }

  private isDuplicate(value: string) : boolean {
    let duplicate = false;
    this.chips.forEach(element => {
      if (element.label.toLowerCase() === value.toLowerCase())
        duplicate = true;
    });

    return duplicate;
  }

  results: string[] = [];

  filter($event: any) {
    if ($event.target.value !== null && $event.target.value.length !== '') {
      this.results = this.suggestions
      .filter(i => i.toLowerCase()
        .includes($event.target.value.toLowerCase())
        && !this.isDuplicate(i)
      )

    } else {
      this.results = [];
    }

    if (this.control) this.onTouch();
  }

  selectValue(value: string) {
    console.log(value)
    this.addChip(value, true);
    this.results = [];
  }

  fill($event: any) {
    if (this.inputRef.nativeElement.value.length === 0)
      this.results = [];
    else this.filter($event);
  }

  @HostListener('document:click', ['$event'])
  onExternalClick($event: any) {
    if (!$event.target.classList.contains('select-option')
    && !this.inputRef.nativeElement.contains($event.target)) {
      this.results = [];
    }

  }
}

  </small>
