import { Component, Input, OnInit, ViewChild, ElementRef, Output, EventEmitter } from '@angular/core';
import { faAngleLeft, faAngleRight, faMapPin, faPencilAlt, faPlus } from '@fortawesome/free-solid-svg-icons';
import { fade } from '../../../animations/shared.animation';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { CalendarEvent, Day } from '../../../models/events';
import { DateService } from '../../../services/date.service';



@Component({
  selector: 'lib-calendar',
  templateUrl: './calendar.component.html',
  styleUrls: ['./calendar.component.scss'],
  animations: [fade]
})
export class CalendarComponent implements OnInit {

  constructor(public dateService: DateService) {}

  ngOnInit(): void {
    this.reset();
  }

  faAngleLeft = faAngleLeft;
  faAngleRight = faAngleRight;
  faMapPin = faMapPin;
  faPlus = faPlus;
  faPencil = faPencilAlt;

  calendarEvents: CalendarEvent[] = [];
  pageIndex = 0;
  selectedDay!: number;
  selectedMonth!: number;
  selectedYear!: number;
  selectedMonthDays: Day[] = [];
  years: string[] = [];
  view = 'days';
  dialog = false;
  editingMode = false;

  @ViewChild('detailsRef') detailsRef!: ElementRef;
  @ViewChild('selectYearRef') selectYearRef: any;
  @ViewChild('selectMonthRef') selectMonthRef: any;

  @Input('class') class: string | null = null;
  @Input('maxDate') maxDate : { year: number, month: number, day: number } = { year: 2050, month: 5, day: 10};
  @Input('minDate') minDate : { year: number, month: number, day: number } =  { year: 2018, month: 1, day: 8};
  @Input('colors') colors: string[] = [
    'crimson',
    'tomato',
    'yellow',
    'aquamarine',
    'cornflowerblue',
    'blueviolet',
  ]

  @Input('events') events: CalendarEvent[] = [
    {
      startDate: new Date(2022, 5, 6, 0, 0, 0, 0),
      endDate: new Date(2022, 5, 10, 0, 0, 0, 0),
      title: 'Meeting',
      color: 'crimson',
      description: 'This is a description, used as an example',
    },
    {
      startDate: new Date(2022, 5, 9, 0, 0, 0, 0),
      endDate: new Date(2022, 5, 18, 0, 0, 0, 0),
      title: 'Meeting',
      color: 'aquamarine',
      description: 'This is a description, used as an example',
    },
  ];

  @Output('onChange') onChange = new EventEmitter();

  increment() {
    if (!(this.selectedYear >= this.maxDate.year
        && this.selectedMonth >= this.maxDate.month)) {
      if (this.selectedMonth === 11) {
        this.selectedMonth = 0;
        this.selectedYear++;
        this.selectYearRef.selectedOption = this.selectedYear;
      } else {
        this.selectedMonth = this.selectedMonth + 1;

      }
      this.selectMonthRef.selectedOption = this.dateService.longMonths[this.selectedMonth];
      this.fallbackDate();
      this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, this.events);
    }
  }

  decrement() {
    if (!(this.selectedYear <= this.minDate.year
      && this.selectedMonth <= this.minDate.month)) {
        if (this.selectedMonth === 0) {
          this.selectedMonth = 11;
          this.selectedYear--;
          this.selectYearRef.selectedOption = this.selectedYear;
        } else {
          this.selectedMonth = this.selectedMonth - 1;
        }
        this.selectMonthRef.selectedOption = this.dateService.longMonths[this.selectedMonth];

        this.fallbackDate();
        this.selectedMonthDays =
      this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
        this.minDate, this.maxDate, this.events);
      }

  }

  getYears(year: number) {
    if (year <= this.maxDate.year && year >= this.minDate.year) {
      this.years = [];
      for (let i = this.currentDate.getFullYear(); i <= this.maxDate.year; i++) {
        this.years.push(i.toString())
      }
    }
  }

  selectMonth($event: any) {
    this.selectedMonth = this.dateService.longMonths.findIndex(m => m === $event);
    this.fallbackDate();
    console.log(this.selectedMonth)
    this.selectedMonthDays =
    this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
      this.minDate, this.maxDate, this.events);
  }

  selectYear($event: any) {
    this.selectedYear = +$event;
    this.fallbackDate();
    this.selectedMonthDays =
    this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
      this.minDate, this.maxDate, this.events);
  }

  reset() {
    let date = new Date();
    this.selectedDay = date.getDate();
    this.selectedMonth = date.getMonth();
    this.selectedYear = date.getFullYear();
    this.fallbackDate();
    this.selectedMonthDays =
    this.dateService.getMonthDays(this.selectedMonth, this.selectedYear,
      this.minDate, this.maxDate, this.events);
    this.getYears(this.selectedYear);
  }

  private fallbackDate() {
    if (this.selectedDay > this.dateService.getDaysCount(this.selectedMonth, this.selectedYear)) {
      this.selectedDay = this.dateService.getDaysCount(this.selectedMonth, this.selectedYear)
    }

    if (this.selectedYear <= this.minDate.year
      && this.selectedMonth <= this.minDate.month
      && this.selectedDay <= this.minDate.day) {
      this.selectedDay = this.minDate.day;
      this.selectedMonth = this.minDate.month;
      this.selectedYear = this.minDate.year;
    }

    if (this.selectedYear >= this.maxDate.year
      && this.selectedMonth >= this.maxDate.month
      && this.selectedDay >= this.maxDate.day) {
        this.selectedDay = this.maxDate.day;
        this.selectedMonth = this.maxDate.month;
        this.selectedYear = this.maxDate.year;
    }

  }

  focusedEvent: CalendarEvent | null = null;
  focusedDay!: Day;

  showEvents(day: Day) {
    this.dialog = true;
  }

  form!: FormGroup;

  addEvent() {
    const event = new CalendarEvent();
    event.color = this.form.get('color')?.value;
    event.title = this.form.get('title')?.value;
    event.description = this.form.get('description')?.value;
    event.startDate = this.form.get('dateRange')?.value.startDate;
    event.endDate = this.form.get('dateRange')?.value.endDate;
    event.startTime = this.form.get('startTime')?.value;
    event.endTime = this.form.get('endTime')?.value;
    this.onChange.emit(event);
    this.dialog = false;
    this.pageIndex = 0;
    this.editingMode = false;
    console.log(event)
  }

  onDialogClose() {
    this.editingMode = false;
    this.pageIndex = 0;
    if (this.form) this.form.reset();
    this.focusedEvent = null;
  }

  getInRange(day: number, event: CalendarEvent) {
    return this.dateService
    .getInRange(day, this.selectedMonth, this.selectedYear, event.startDate, event.endDate)
  }

  setFocusedEvent($event?: CalendarEvent) {
    if ($event) this.focusedEvent = $event;
    this.dialog = true;
    this.editingMode = true;
    this.form = new FormGroup({
      title: new FormControl(this.focusedEvent?.title, Validators.required),
      description: new FormControl(this.focusedEvent !== null? this.focusedEvent.description : null, Validators.required),
      dateRange: new FormControl(this.focusedEvent?
        { startDate: this.focusedEvent.startDate, endDate: this.focusedEvent.endDate }
      : { startDate: {day: this.selectedDay, month: this.selectedMonth, year: this.selectedYear},
          endDate: {day: this.selectedDay, month: this.selectedMonth, year: this.selectedYear}
        }, Validators.required),
      startTime: new FormControl(this.focusedEvent? this.focusedEvent.startTime : 'All Day', Validators.required),
      endTime: new FormControl(this.focusedEvent? this.focusedEvent.endTime : 'All Day', Validators.required),
      color: new FormControl(this.focusedEvent? this.focusedEvent.color : null, Validators.required),
      location: new FormControl(this.focusedEvent? this.focusedEvent.location : null, Validators.required),
      attachment: new FormControl(this.focusedEvent? this.focusedEvent.attachment: null, Validators.required)
    })
  }

  currentDate = new Date();

  // orderby time
  // make view for time
}
