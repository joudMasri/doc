  getSpace(j: number, strokeWidth: number) {
    if (this.data.datasets.length % 2 === 0) {
      let compared = this.data.datasets.length/2;
      if (j < compared) {
        return -((compared - j) * strokeWidth - strokeWidth/2);
      } else if (j === compared) {
        return strokeWidth/2;
      } else {
        return (j - compared) * strokeWidth  + strokeWidth/2;
      }
    } else {
      let compared = Math.floor(this.data.datasets.length/2);
      if (j < compared) {
        return (-j - 1) * strokeWidth;
      } else if (j === compared) {
        return 0;
      } else {
        return ((j) - compared) * strokeWidth;
      }
    }
  }

  transformValuesToPercent() {
    this.pieData = this.pieData.sort((a,b) => a.data - b.data);
    let cumulativePercent = 0;

    this.pieData.forEach(element => {
      element.percent = element.data/this.total;
      element.startX = this.getCoordinatesForPercent(cumulativePercent)[0];
      element.startY = this.getCoordinatesForPercent(cumulativePercent)[1]

      cumulativePercent += element.percent;

      element.endX = this.getCoordinatesForPercent(cumulativePercent)[0];
      element.endY = this.getCoordinatesForPercent(cumulativePercent)[1];
    });

  }

  getCoordinatesForPercent(percent: number, radius: number = this.radius) {
    const x = Math.cos(2 * Math.PI * percent) * radius;
    const y = Math.sin(2 * Math.PI * percent) * radius;
    return [x, y];
  }
